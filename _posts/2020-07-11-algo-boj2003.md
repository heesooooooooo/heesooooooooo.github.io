	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan=new Scanner(System.in);
		int n=scan.nextInt();
		int m=scan.nextInt();
		int[] array=new int[n];
		for(int i=0;i<n;i++)
			array[i]=scan.nextInt();
		
		int cnt=0;
		for(int i=0;i<n;i++) {
			int sum=0;
			for(int j=i;j<n;j++) {
				sum+=array[j];
				if(sum==m) {
					cnt++;
					break;
				}
			}
		}
		
		System.out.println(cnt);
	}
}
