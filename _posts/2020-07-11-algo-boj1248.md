 * @author HEESOO

public class Main {
	static int n;
	static char[][] array;
	static int[] result;
	public static void dfs(int cnt) {
		if(cnt==n) {
			for(int i=0;i<n;i++)
				System.out.print(result[i]+" ");
			System.exit(0);
			return;
		}
		
		for(int i=-10;i<=10;i++) {
			result[cnt]=i;
			if(check(cnt)) dfs(cnt+1);
		}
	}
	public static boolean check(int idx) {
		for(int i=0;i<=idx;i++) {
			int sum=0;
			for(int j=i;j<=idx;j++) {
				sum+=result[j];
				if(array[i][j]!=(sum==0?'0':(sum>0?'+':'-')))
					return false;
			}
		}
		return true;
	}
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan=new Scanner(System.in);
		n=scan.nextInt();
		String str=scan.next();
		array=new char[n][n];
		int idx=0;
		for(int i=0;i<n;i++) {
			for(int j=i;j<n;j++) {
				array[i][j]=str.charAt(idx);
				idx++;
			}				
		}			
		result=new int[n];
		
		dfs(0);
	}
}
