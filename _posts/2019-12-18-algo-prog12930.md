 class Solution {
  public String solution(String s) {
      String answer = "";
      String[] array=s.split("");
      int blank=0;
      for(String str:array){
          if(str.equals(" ")){
              answer+=" ";
              blank=0;
              continue;
          }
          if(blank%2==0)
              answer+=str.toUpperCase();
          else
              answer+=str.toLowerCase();
          blank++;
      }
      return answer;
  }
}
