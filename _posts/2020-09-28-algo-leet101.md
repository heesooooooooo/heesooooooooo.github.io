/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isSymmetric(TreeNode root) {
        Queue<TreeNode> q=new LinkedList<>();
        q.offer(root);
        q.offer(root);
        
        while(!q.isEmpty()){
            TreeNode a=q.poll();
            TreeNode b=q.poll();
            // ë‘ ê°œ ë¹„êµ
            if(a==null && b==null) continue;
            if(a==null || b==null) return false;
            if(a.val!=b.val) return false;
            // ë‘ ê°’ì´ ëŒ€ì¹­ì´ë¼ë©´ ìˆœì„œëŒ€ë¡œ ë„£ê¸°
            q.offer(a.left);
            q.offer(b.right);
            q.offer(a.right);
            q.offer(b.left);
        }
        
        return true;
    }
}
