class Main {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int T=sc.nextInt();
		for(int t=0;t<T;t++) {
        	int N=sc.nextInt();
        	int M=sc.nextInt();
        	
        	int[][] dp=new int[30][30];
        	dp[1][0]=dp[1][1]=1;
        	for(int i=2;i<=M;i++) {
        		for(int j=0;j<=i;j++) {
        			if(j==0 || i==j) dp[i][j]=1;
        			else dp[i][j]=dp[i-1][j]+dp[i-1][j-1];
        		}
        	}
        	
        	System.out.println(dp[M][N]);
        }
	}
	

}
