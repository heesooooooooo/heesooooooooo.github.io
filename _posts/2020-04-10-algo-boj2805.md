import java.util.Arrays;
import java.util.Scanner;

/**
 * @author HEESOO
 *
 */
public class Main {
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan = new Scanner(System.in);
		int n=scan.nextInt();
		int m=scan.nextInt();
		int[] array=new int[n];
		for(int i=0;i<n;i++)
			array[i]=scan.nextInt();
		Arrays.sort(array);
		
		int left=0;
		int right=array[n-1];
		int mid=0;
		int max=0;
		while(left<=right) {
			long meter=0;
			mid=(left+right)/2;
			for(int i=0;i<n;i++) {
				if(mid<array[i])
					meter+=array[i]-mid;
			}
			
			if(meter>=m) {
				left=mid+1;
				max=Math.max(max, mid);
			}
			else
				right=mid-1;
		}
		
		System.out.println(max);		
		
	}
}
