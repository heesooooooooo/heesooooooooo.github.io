public class Main {
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan=new Scanner(System.in);
		int n=scan.nextInt();
		int[] array=new int[n];
		int[] dp=new int[n];
		int max=0;
		
		for(int i=0;i<n;i++)
			array[i]=scan.nextInt();
				
		for(int i=0;i<n;i++) {
			dp[i]=1;
			for(int j=0;j<i;j++) {
				if(array[i]<array[j] && dp[i]<dp[j]+1)
					dp[i]=dp[j]+1;
			}
			max=Math.max(max, dp[i]);
		}
		
		System.out.println(max);
		
	}

}
