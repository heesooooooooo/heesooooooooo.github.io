	public static void main(String[] args){
		Scanner input=new Scanner(System.in);
		int n=input.nextInt();
		int[] array=new int[n];
		for(int i=0;i<n;i++){
			array[i]=input.nextInt();
		}
		
		int[] dp=new int[n];
		dp[0]=array[0];
		
		int max=array[0];
		for(int i=1;i<n;i++){
			dp[i]=Math.max(array[i], dp[i-1]+array[i]);
			max=Math.max(max, dp[i]);
		}
		
		System.out.println(max);
		
	}
}
