 class Solution {
   public String solution(String s, int n) {
       String answer = "";
       char alpha;
       for(int i=0;i<s.length();i++){
           alpha=s.charAt(i);
           if(alpha==' ') answer+=" ";
           else if('A'<=alpha&&alpha<='Z'){
               if(alpha+n>'Z'){
                   answer+=(char)(alpha+n-26);
               }
               else{
                   answer+=(char)(alpha+n);
               }
           }
           else{
              if(alpha+n>'z'){
                  answer+=(char)(alpha+n-26);
              }
              else{
                  answer+=(char)(alpha+n);
              }
           }
       }
       return answer;
   }
 }
