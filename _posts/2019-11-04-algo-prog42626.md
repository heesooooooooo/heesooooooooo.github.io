import java.util.PriorityQueue;
/**
 *
 * @author HEESOO
 *
 */
 class Solution {
     public int solution(int[] scoville, int K) {
         int answer = 0;
         int a,b;
         PriorityQueue<Integer> pq=new PriorityQueue<>();
         for(int i:scoville){
             pq.offer(i);
         }
         while(pq.peek()<K){
             if(pq.size()==1){
                 return -1;
             }
             else{
                 a=pq.poll();
                 b=pq.poll();
                 pq.offer(a+(b*2));
                 answer++;
             }
         }
         return answer;
     }
 }
