class Solution {
    public void rotate(int[][] matrix) {
        int n=matrix.length;
        int top=0, down=n-1;
        while(top<down){ // ìœ„ì™€ ì•„ë˜ í–‰ë ¬ì„ swap, ë‘˜ì´ ë§Œë‚  ë•Œ ê¹Œì§€ ë°˜ë³µ
            int[] temp=matrix[top];

            matrix[down]=temp;
            top++;
            down--;
        }
        
        for(int i=0;i<n;i++){ // ëŒ€ê°ì„  ê¸°ì¤€ ëŒ€ì¹­ swap

            }
        }
    }
}
