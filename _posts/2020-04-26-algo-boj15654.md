import java.util.Arrays;
import java.util.Scanner;

/**
 * @author HEESOO

 */
public class Main {
	static int n, m;
	static int[] array;
	static int[] result;
	static boolean[] visit;
	static StringBuilder sb;
	public static void dfs(int idx) {
		if(idx==m) {
			for(int i=0;i<m;i++)
				sb.append(result[i]+" ");
			sb.append("\n");
			return;
		}
		for(int i=0;i<n;i++) {
			if(visit[i]) continue;
			visit[i]=true;
			result[idx]=array[i];
			dfs(idx+1);
			visit[i]=false;
		}
	}
	
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan = new Scanner(System.in);
		n=scan.nextInt();
		m=scan.nextInt();
		array=new int[n];
		result=new int[m];
		visit=new boolean[n];
		sb=new StringBuilder();
		for(int i=0;i<n;i++)
			array[i]=scan.nextInt();
		
		Arrays.sort(array);
		dfs(0);
		System.out.println(sb);
	}
}
