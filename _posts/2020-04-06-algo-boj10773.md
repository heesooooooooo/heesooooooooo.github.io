import java.util.Scanner;
import java.util.Stack;
/**
 * @author HEESOO

 */
public class Main {
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner input=new Scanner(System.in);
		int k=input.nextInt();
		int answer=0;
		int num;
		Stack<Integer> st=new Stack<>();
		
		for(int i=0;i<k;i++) {
			num=input.nextInt();
			if(num==0)
				answer-=st.pop();
			else {
				answer+=num;
				st.push(num);
			}				
		}
		
		System.out.println(answer);
	}
}
