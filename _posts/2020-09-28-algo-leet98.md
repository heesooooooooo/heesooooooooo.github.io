 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    List<Integer> list;
    public boolean isValidBST(TreeNode root) {
        if(root==null) return true;
        
        list=new ArrayList<>();
        inorder(root);
        
        for(int i=0;i<list.size()-1;i++){
            // ë’¤ì— ë‚˜ë³´ë‹¤ ê°™ê±°ë‚˜ ë” í° ê°’ì´ ìˆìœ¼ë©´
            if(list.get(i)>=list.get(i+1)) return false;
        }
        
        return true;
    }
    
    public void inorder(TreeNode node){ // ì¤‘ìœ„ìˆœíšŒ
        if(node!=null){
            inorder(node.left);
            list.add(node.val);
            inorder(node.right);
        }
    }
}
