class Solution {
    public int majorityElement(int[] nums) {
        int n=nums.length;
        
        HashMap<Integer, Integer> map=new HashMap<>();
        for(int num:nums){ // numì˜ ê°œìˆ˜ ì¹´ìš´íŠ¸
            int value=map.getOrDefault(num, 0);
            map.put(num, value+1);
        }
        
        int answer=0;
        int majority=n/2;
        for(Map.Entry<Integer, Integer> entry:map.entrySet()){
            if(entry.getValue()>majority) { // majority ë„˜ëŠ” valueì˜ key ë¦¬í„´
                answer=entry.getKey();
                break;
            }            
        }
        
        return answer;
    }
