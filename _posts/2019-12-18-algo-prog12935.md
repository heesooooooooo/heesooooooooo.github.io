 * @author HEESOO

 class Solution {
   public int[] solution(int[] arr) {
       int[] answer = {};
       if(arr.length==1){
           return new int[] {-1};
       }
       else{
           int min=arr[0];
           for(int num:arr){
               if(min>num) min=num;
           }
           answer=new int[arr.length-1];
           int i=0;
           for(int num:arr){
               if(num!=min){
                   answer[i]=num;
                   i++;
               }
           }
       }
       return answer;
   }
 }
