		int n=input.nextInt();
		int m=input.nextInt();
		int[] card=new int[n];
		for(int i=0;i<n;i++){
			card[i]=input.nextInt();
		}
		
		int sum;
		int sub=m, tempSum=0;
		for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				for(int k=j+1;k<n;k++){
					sum=card[i]+card[j]+card[k];
					if(m==sum){
						System.out.println(sum);
						return;
					}
					else if(m>sum){
						if(sub>m-sum){
							sub=m-sum;
							tempSum=sum;
						}
					}
				}
			}
		}
		System.out.println(tempSum);
		
	}
	
}
