class Solution {

    public int solution(int[] nums) {

        Arrays.sort(nums);
        int max=nums[nums.length-1]; // numsμ—μ„ μ μΌ ν° κ°’

            if(!prime[i]) continue;

                if(!prime[j]) continue;
                prime[j]=false;
            }
        }
        
        // μ΅°ν•©μΌλ΅ 3κ° μ„ νƒ
