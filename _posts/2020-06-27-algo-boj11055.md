	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan=new Scanner(System.in);
		int n=scan.nextInt();
		int[] array=new int[n];
		int[] dp=new int[n];
		int max=0;
		
		for(int i=0;i<n;i++)
			array[i]=scan.nextInt();
				
		for(int i=0;i<n;i++) {
			dp[i]=array[i];
			for(int j=0;j<i;j++) {
				if(array[j]<array[i] && dp[i]<dp[j]+array[i])
					dp[i]=dp[j]+array[i];
			}
			max=Math.max(max, dp[i]);
		}
		
		System.out.println(max);
		
	}

}
