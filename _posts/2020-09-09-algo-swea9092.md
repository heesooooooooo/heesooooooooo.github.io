class Main {
	static int n, k;
	static int[][] dot, dp;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int T;
		T=sc.nextInt();
		
		for(int test_case = 1; test_case <= T; test_case++)
		{	
			n=sc.nextInt();
			k=sc.nextInt();
			dot=new int[n][2];
			for(int i=0;i<n;i++) {
				dot[i][0]=sc.nextInt();
				dot[i][1]=sc.nextInt();
			}
			
			dp=new int[n][k+1];
			for(int[] i:dp) Arrays.fill(i, Integer.MAX_VALUE);
			int answer=solve(0,k);
			
			System.out.println("#"+test_case+" "+answer);
		}
	}
	
	public static int solve(int cur, int k) {
		if(cur==n-1) return 0; // ë„ì°©
		if(dp[cur][k]!=Integer.MAX_VALUE) return dp[cur][k]; // ì¤‘ë³µ ê³„ì‚° ë°©ì§€
		
		for(int i=0;i<=k;i++) {
			int next=cur+i+1; // iì¹¸ ë›°ê¸°
			if(next>=n) continue; //next ë²”ìœ„ ì²´í¬
			if(k-i<0) continue; 
			
			int dist=Math.abs(dot[cur][0]-dot[next][0])+Math.abs(dot[cur][1]-dot[next][1]);
			dp[cur][k]=Math.min(dp[cur][k], solve(next, k-i)+dist); 
		}
		
		return dp[cur][k];
	}
}
