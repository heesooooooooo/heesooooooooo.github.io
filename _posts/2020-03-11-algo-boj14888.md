import java.util.Scanner;

/**
 * 
 * @author HEESOO

public class Main {
	static int[] number;
	static int[] operator;
	static int min=Integer.MAX_VALUE;
	static int max=Integer.MIN_VALUE;
	
	public static void dfs(int idx, int sum){
		if(idx==number.length){
			if(min>sum) min=sum;
			if(max<sum) max=sum;
			return;
		}
		if(operator[0]>0){
			operator[0]--;
			dfs(idx+1, sum+number[idx]);
			operator[0]++;
		}
		if(operator[1]>0){
			operator[1]--;
			dfs(idx+1, sum-number[idx]);
			operator[1]++;
		}
		if(operator[2]>0){
			operator[2]--;
			dfs(idx+1, sum*number[idx]);
			operator[2]++;
		}
		if(operator[3]>0){
			operator[3]--;
			dfs(idx+1, sum/number[idx]);
			operator[3]++;
		}
	}
	public static void main(String[] args){
		Scanner input=new Scanner(System.in);
		int n=input.nextInt();
		number=new int[n];
		operator=new int[4];
		for(int i=0;i<n;i++){
			number[i]=input.nextInt();
		}
		for(int i=0;i<4;i++){
			operator[i]=input.nextInt();
		}
		
		dfs(1, number[0]);
		System.out.println(max);
		System.out.println(min);
		
	}
}
