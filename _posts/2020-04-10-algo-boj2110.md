import java.util.Arrays;
import java.util.Scanner;

/**
 * @author HEESOO

 */
public class Main {
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan = new Scanner(System.in);
		int n=scan.nextInt();
		int c=scan.nextInt();
		int[] array=new int[n];
		for(int i=0;i<n;i++)
			array[i]=scan.nextInt();
		Arrays.sort(array);
		
		int left=1;
		int right=array[n-1]-array[0];
		int mid=0;
		int max=0;
		while(left<=right) {
			mid=(left+right)/2;
			
			int cnt=1;
			int install=array[0];
			for(int i=1;i<n;i++) {
				if(install+mid<=array[i]) {
					cnt++;
					install=array[i];
				}
			}
			
			if(cnt>=c) {
				max=Math.max(max, mid);
				left=mid+1;
			}
			else 
				right=mid-1;
		}
		
		System.out.println(max);
	}
}
