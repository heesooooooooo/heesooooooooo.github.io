	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan = new Scanner(System.in);
		int n=scan.nextInt();
		int[] array=new int[n];
		for(int i=0;i<n;i++)
			array[i]=scan.nextInt();
		Arrays.sort(array);
		
		int m=scan.nextInt();
		int[] input=new int[m];
		for(int i=0;i<m;i++)
			input[i]=scan.nextInt();
		
		for(int i=0;i<m;i++) {
			int first=0;
			int last=n-1;
			int mid=0;
			while(first<=last) {
				mid=(first+last)/2;
				if(input[i]==array[mid]) {
					System.out.println("1");
					break;
				}
				else {
					if(input[i]<=array[mid]) 
						last=mid-1;
					else
						first=mid+1;
				}
			}
			if(first>last)
				System.out.println("0");
		}
	}
}
