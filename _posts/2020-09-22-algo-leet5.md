class Solution {
    public String longestPalindrome(String s) {
        if(s.length()==0 || s==null) return "";
        int n=s.length();
        boolean[][] dp=new boolean[n][n];
        int maxSize=0;
        String answer="";
        
        for(int j=0;j<n;j++){// ì¢…ë£Œ ì¸ë±ìŠ¤
            for(int i=0;i<=j;i++){ // ì‹œì‘ ì¸ë±ìŠ¤
                if(i==j) dp[i][j]=true;//í•œ ìë¦¬ì¸ ê²½ìš°
                else if(j==i+1){// ë‘ìë¦¬ì¸ ê²½ìš°
                    if(s.charAt(i)==s.charAt(j)) dp[i][j]=true; // ì„œë¡œ ê°™ì•„ì•¼ T
                }
                else{
                    boolean same=s.charAt(i)==s.charAt(j); //i, jê°€ ê°™ì€ì§€ ì²´í¬
                    dp[i][j]=dp[i+1][j-1] && same; //i+1~j-1ê°€ Tì¸ì§€ ì²´í¬

        }
        
        
        return answer;
    }
    
    
}
