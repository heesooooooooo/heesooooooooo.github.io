import java.util.Arrays;
import java.util.Scanner;

/**
 * @author HEESOO
 *
 */
public class Main {
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan = new Scanner(System.in);
		int n=scan.nextInt();
		int k=scan.nextInt();
		
		int left=1;
		int right=k;
		int mid=0;
		int answer=0;
		while(left<=right) {
			mid=(left+right)/2;
			int cnt=0;
			for(int i=1;i<=n;i++)
				cnt+=Math.min(mid/i, n);
			
			if(cnt>=k) {
				answer=mid;
				right=mid-1;
			}
			else
				left=mid+1;
		}
		
		System.out.println(answer);
	}
}
