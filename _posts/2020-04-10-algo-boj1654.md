import java.util.Arrays;
import java.util.Scanner;

/**
 * @author HEESOO
 *
 */
public class Main {
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan = new Scanner(System.in);
		int k=scan.nextInt();
		int n=scan.nextInt();
		int[] array=new int[k];
		for(int i=0;i<k;i++)
			array[i]=scan.nextInt();
		Arrays.sort(array);
		
		long left=1;
		long right=array[k-1];
		long mid=0;
		long max=0;
		
		while(left<=right) {
			mid=(left+right)/2;
			long cnt=0;
			for(int i=0;i<k;i++) 
				cnt+=array[i]/mid;
			
			if(cnt>=n) {
				left=mid+1;
				max=Math.max(max, mid);
			}
			else {
				right=mid-1;
			}
		}
		
		System.out.println(max);
		
	}
}
