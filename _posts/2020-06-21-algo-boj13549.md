public class Main {
	public static int bfs(int n, int k) {
		Queue<Integer> q=new LinkedList<>();
		int[] visit=new int[100001];
		Arrays.fill(visit, -1);
		q.offer(n);
		visit[n]=0;
		
		while(!q.isEmpty()) {
			int now=q.poll(); //λ‚΄ μ„μΉ
			if(now==k) return visit[now]; // λ™μƒ μ°Ύμ
			
			// μκ°„μ΄λ™ν•  μ μλ” κ³³ λ‹¤ μ°ΎκΈ°
			int temp=now*2;
			while(temp<=100000 && visit[temp]==-1) {
				visit[temp]=visit[now];
				q.offer(temp);
				temp*=2;
			}
			
			// 1μ΄ ν›„
			if(0<=now-1 && visit[now-1]==-1) {
				visit[now-1]=visit[now]+1;
				q.offer(now-1);
			}
			if(now+1<=100000 && visit[now+1]==-1) {
				visit[now+1]=visit[now]+1;
				q.offer(now+1);
			}
			
		}
		return 0;
	}
	public static void main(String[] args) {

		int n=scan.nextInt();

	}
