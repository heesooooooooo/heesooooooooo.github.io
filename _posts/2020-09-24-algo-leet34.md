    public int[] searchRange(int[] nums, int target) {
        int mid=binarySearch(nums, target); // target í•˜ë‚˜ ì°¾ê¸°
        if(mid==-1) return new int[]{-1,-1}; // ì—†ì„ ê²½ìš°
        
        int[] answer=new int[2];
        
        int left=mid;
        while(0<=left){ // ì™¼ìª½ìœ¼ë¡œ ê³„ì† ì´ë™
            if(nums[left]==target) answer[0]=left; // ê°’ ê°±ì‹ 
            else break; // targetì´ ì•„ë‹ˆë¼ë©´ ì¢…ë£Œ
            left--;
        }
        
        int right=mid;
        while(right<nums.length){ // ì˜¤ë¥¸ìª½ìœ¼ë¡œ ì´ë™
            if(nums[right]==target) answer[1]=right;
            else break;
            right++;
        }        
        
        return answer;
    }
    
    public int binarySearch(int[] nums, int target){ // ì´ë¶„íƒìƒ‰
        int left=0, right=nums.length-1;
        while(left<=right){
            int mid=(left+right)/2;
            if(nums[mid]==target) return mid;
            else if(nums[mid]<target) left=mid+1;
            else right=mid-1;
        }
        
        return -1;
    }
}
