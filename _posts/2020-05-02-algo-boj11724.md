import java.util.Scanner;

/**
 * @author HEESOO
 *
 */
public class Main {
	static int n, m;
	static int[][] array;
	static boolean[] visit;
	public static void dfs(int idx) {
		for(int i=1;i<=n;i++) {
			if(visit[i]) continue;
			if(array[idx][i]==1) {
				visit[i]=true;
				dfs(i);
			}
		}
	}
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner scan = new Scanner(System.in);
		n=scan.nextInt();
		m=scan.nextInt();
		array=new int[n+1][n+1];
		visit=new boolean[n+1];
		for(int i=0;i<m;i++) {
			int a=scan.nextInt();
			int b=scan.nextInt();
			array[a][b]=1;
			array[b][a]=1;
		}
		
		int cnt=0;
		for(int i=1;i<=n;i++) {
			if(visit[i]) continue;
			cnt++;
			dfs(i);
		}
		
		System.out.println(cnt);
	}
}
