/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int maxDepth(TreeNode root) {
        int depth=-1;
        Queue<TreeNode> q=new LinkedList<>();
        q.offer(root);
        
        while(!q.isEmpty()){
            int size=q.size();
            depth++; // í˜„ì¬ ë ˆë²¨ ê³„ì‚°
            for(int i=0;i<size;i++){ // ë ˆë²¨ì— ìˆëŠ” ë…¸ë“œ ì²´í¬
                TreeNode node=q.poll();
                if(node==null) continue;
                q.offer(node.left);
                q.offer(node.right);
            }
        }
        
        return depth;
    }
}
