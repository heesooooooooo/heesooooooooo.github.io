    public int[] solution(int[] heights) {
        int[] answer = {};
        Stack<Top> st=new Stack<Top>();
        Stack<Top> stTemp=new Stack<Top>();

                answer[i]=0;

                now=st.pop();
                next=st.pop();

            while(now.height>=next.height){//nextì¡°ê±´ì— ì•ˆë§ìŒ
