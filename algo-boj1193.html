 * 
 * @author HEESOO

public class Main {
	public static void main(String[] args) {
		Scanner input=new Scanner(System.in);
		int n=input.nextInt();
		
		int line=1;
		int cnt=0;
		while(true){
			if(cnt+line>=n) break;
			cnt+=line;
			line++;
		}
		int th=n-cnt;
		
		if(line%2==0){
			System.out.println(th+"/"+(line-th+1));
		}
		else{
			System.out.println((line-th+1)+"/"+th);
		}
	}
}
